def _generate_name_variants(self, original_name: str) -> List[str]:
    """
    –ì–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç –≤—Å–µ –≤–æ–∑–º–æ–∂–Ω—ã–µ –≤–∞—Ä–∏–∞–Ω—Ç—ã –∏–º–µ–Ω–∏ –¥–ª—è —à–∞–±–ª–æ–Ω–∞
    
    –ü—Ä–∏–º–µ—Ä –¥–ª—è "Chrome-TikTok-like-btn.png":
    - Chrome-TikTok-like-btn    (–æ—Ä–∏–≥–∏–Ω–∞–ª)
    - TikTok-like-btn          (–±–µ–∑ Chrome-)
    - like-btn                 (–±–µ–∑ Chrome-TikTok-)
    - like                     (–±–µ–∑ -btn)
    - like_btn                 (–¥–µ—Ñ–∏—Å—ã ‚Üí –ø–æ–¥—á–µ—Ä–∫–∏–≤–∞–Ω–∏—è)
    - TikTokLike               (CamelCase)
    """
    variants = [original_name]  # –û—Ä–∏–≥–∏–Ω–∞–ª—å–Ω–æ–µ –∏–º—è
    clean_name = original_name
    
    # –£–±–∏—Ä–∞–µ–º –ø—Ä–µ—Ñ–∏–∫—Å—ã
    prefixes_to_remove = [
        "Chrome-TikTok-", "Chrome-YouTube-", "Chrome-Instagram-",
        "Chrome-", "Atlas-", "Safari-", "System-"
    ]
    
    for prefix in prefixes_to_remove:
        if clean_name.startswith(prefix):
            clean_name = clean_name[len(prefix):]
            variants.append(clean_name)
            break
    
    # –£–±–∏—Ä–∞–µ–º —Å—É—Ñ—Ñ–∏–∫—Å—ã
    suffixes_to_remove = ["-btn", "_btn", "-button", "_button"]
    for suffix in suffixes_to_remove:
        if clean_name.endswith(suffix):
            short_name = clean_name[:-len(suffix)]
            variants.append(short_name)
    
    # –í–∞—Ä–∏–∞–Ω—Ç—ã —Å –∑–∞–º–µ–Ω–æ–π —Å–∏–º–≤–æ–ª–æ–≤
    for variant in variants.copy():
        # –î–µ—Ñ–∏—Å—ã ‚Üí –ø–æ–¥—á–µ—Ä–∫–∏–≤–∞–Ω–∏—è
        if "-" in variant:
            variants.append(variant.replace("-", "_"))
        
        # CamelCase –≤–µ—Ä—Å–∏—è
        camel_case = "".join(word.capitalize() for word in variant.split("-"))
        if camel_case != variant:
            variants.append(camel_case)
    
    return list(set(variants))  # –£–±–∏—Ä–∞–µ–º –¥—É–±–ª–∏–∫–∞—Ç—ã
```

def _parse_click_command(self, line: str) -> Dict[str, Any]:
    """
    –ü–∞—Ä—Å–∏—Ç –∫–æ–º–∞–Ω–¥—É click –∏ –Ω–∞—Ö–æ–¥–∏—Ç —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–π —à–∞–±–ª–æ–Ω

    1. –ò–∑–≤–ª–µ–∫–∞–µ—Ç –∏–º—è —ç–ª–µ–º–µ–Ω—Ç–∞ –∏–∑ –∫–æ–º–∞–Ω–¥—ã
    2. –ò—â–µ—Ç –≤ template_map
    3. –ï—Å–ª–∏ –Ω–∞–π–¥–µ–Ω ‚Üí —Å–æ–∑–¥–∞–µ—Ç click_template –∫–æ–º–∞–Ω–¥—É
    4. –ï—Å–ª–∏ –Ω–µ –Ω–∞–π–¥–µ–Ω ‚Üí –æ—à–∏–±–∫–∞ –∏–ª–∏ fallback
    """
    # –ò–∑–≤–ª–µ–∫–∞–µ–º –∏–º—è —ç–ª–µ–º–µ–Ω—Ç–∞
    match = re.match(r'click\s+(.+)', line.strip())
    if not match:
        return None
    
    element_name = match.group(1).strip()
    
    # –ò—â–µ–º –≤ –∫–∞—Ä—Ç–µ —à–∞–±–ª–æ–Ω–æ–≤
    template_path = self.template_map.get(element_name)
    
    if template_path:
        return {
            'action': 'click_template',
            'template': template_path,
            'confidence': 0.8,
            'timeout': 10,
            'original_name': element_name
        }
    else:
        # –®–∞–±–ª–æ–Ω –Ω–µ –Ω–∞–π–¥–µ–Ω
        print(f"‚ö†Ô∏è –®–∞–±–ª–æ–Ω '{element_name}' –Ω–µ –Ω–∞–π–¥–µ–Ω")
        return {
            'action': 'error',
            'message': f"Template '{element_name}' not found",
            'available_templates': list(self.template_map.keys())[:10]
        }
```
